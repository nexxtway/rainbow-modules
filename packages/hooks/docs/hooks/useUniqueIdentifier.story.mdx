# Overview

`useUniqueIdentifier` A hook that elegantly generate unique identifiers.

### API

```ts
    useUniqueIdentifier(
        prefix: string
    ): string;
```

### Params

| Property | Description                         | Type   | Default   |
| -------- | ----------------------------------- | ------ | --------- |
| prefix   | Base to generate unique identifiers | string | undefined |

### Usage

```js
import React from 'react';
import { useUniqueIdentifier } from '../../src';

export default function WrapperUniqueIdentifier() {
    const uniqueId = useUniqueIdentifier('rainbow');
    return (
        <p id={uniqueId}>
            <span>My unique identifier is: </span>
            {uniqueId}
        </p>
    );
}
```
