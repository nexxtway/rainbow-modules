# Overview

`useOutsideClick` A hook that elegantly manages click outside of target elements.

### API

```ts
    useOutsideClick(
        ref : React.MutableRefObject,
        handle: (event: MouseEvent | TouchEvent) => void,
        isListening: boolean
    );
```

### Params

| Property    | Description      | Type                   | Default |
| ----------- | ---------------- | ---------------------- | ------- |
| ref         | Ref Object       | React.MutableRefObject |         |
| handle      | Trigger Function | (event) => void        |         |
| isListening |                  | boolean                | true    |

### Usage

```js
import React, { useState, useRef } from 'react';
import { useOutsideClick } from '../../src';

export default function WrapperOutsideClik() {
    const [counter, setCounter] = useState(0);
    const [isListening, setIsListening] = useState(true);
    const ref = useRef();
    useOutsideClick(
        ref,
        () => {
            setCounter((s) => s + 1);
        },
        isListening,
    );
    return (
        <div>
            <div>
                <input
                    id="outside-click-input-1"
                    type="checkbox"
                    checked={isListening}
                    onChange={(event) => setIsListening(event.target.checked)}
                />
                <label htmlFor="outside-click-input-1">isListening</label>
            </div>
            <br />
            <span ref={ref}>
                <button type="button">Click Me and Click Outside Me</button>
            </span>
            <p>
                counter:
                {counter}
            </p>
        </div>
    );
}
```
