# Overview

`useFloatingBarScrollHandler` A hook that manages the visibility of `FloatingBar` based on scroll offset.

### API

```ts
useFloatingBarScrollHandler({
    scrollTop: number,
});
```

### Params

| Property  | Description                      | Type   | Default |
| --------- | -------------------------------- | ------ | ------- |
| scrollTop | Scroll offset to show the topbar | number | 50      |

### Usage

```jsx
import React, { useState } from 'react';
import { Application } from 'react-rainbow-components';
import styled from 'styled-components';
import { FloatingBar, useFloatingBarScrollHandler } from '@rainbow-modules/layout';

const Container = styled.div`
    width: 100%;
    height: 200px;
    position: relative;
`;

const ScrollableContainer = styled.div`
    height: 500px;
    overflow: auto;
`;

export const FloatingBarWithHook = () => {
    const [showBar, handleScrollEvent] = useTopBarScrollHandler({
        scrollTop: 50,
    });
    return (
        <Application>
            <Container>
                <FloatingBar isVisible={showBar}>Top bar content goes here</FloatingBar>
                <ScrollableContainer onScroll={handleScrollEvent}>
                    Scrollable content goes here
                </InnerContainer>
            </Container>
        </Application>
    );
};
```
